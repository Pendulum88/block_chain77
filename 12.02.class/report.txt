const TYPE = {
LOGIN: "user/login",
LOGOUT: "user/logout",
REGIST: "user/regist",
}

const login = (aug) => {
type: TYPE.LOGIN,
payload: {aug}
}
const logout = (aug)=> {
type: TYPE.LOGOUT
payload: {aug}
}
const regist = (aug) => {
type: TYPE.REGIST
payload: {aug}
}

export const action = {login, logout, regist}

export const initialize = {
id: "",
pw: "";
}

export const reducer = (state = initialize, action) => {
const {type, payload} = action

switch() {
case: TYPE.LOGIN:
return state

case: TYPE.LOGOUT:
return state

case: TYPE.REGIST:
return state

default:
return state
}
}
